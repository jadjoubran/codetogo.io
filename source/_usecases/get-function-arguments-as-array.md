```javascript
function log(...msgs) {
  //msgs is now an array
  console.log(`Received ${msgs.length} messages`);
}

log("code", "to", "go");
log("code", "to", "go", "is", "cool");
```

<pre class="output">
Received 3 messages
Received 5 messages
</pre>
